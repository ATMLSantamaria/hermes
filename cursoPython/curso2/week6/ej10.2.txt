#!/usr/bin/python3

name = input("Enter file:")
if len(name) < 1 : name = "mbox-short.txt"
handle = open(name)
horas=list()
for line in handle:
	if not line.startswith("From "):
		continue
	line=line.rstrip()
	words=line.split()
	a=words[5]
	tiempo=words[5].split(":")
	horas.append(tiempo[0])	
#print(horas)
count=dict()

for hora in horas:
        count[hora]=count.get(hora,0)+1
lst=list()
for k,v in count.items():
	new=(k,v)
	lst.append(new)

#print(lst)
lst=sorted(lst)
#print(lst)
for key,value in lst:
	print(key,value)
